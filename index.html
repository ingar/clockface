Standalone clock face renderer

var clockRenderer = new clockface.SBBCanvasRenderer({
  el: '#clock-renderer-example',
  secondHand: true
})
function setClockTime() {
  var now = new Date()
  clockRenderer.setTime(now.getHours(), now.getMinutes(), now.getSeconds())
  setTimeout(setClockTime, 500)
}
